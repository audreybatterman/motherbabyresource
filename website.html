function switchTab(name) {

  ['maternal','infant','directory'].forEach(t => {

    const btn   = document.getElementById('tab-'+t);

    const panel = document.getElementById('panel-'+t);

    const active = (t === name);

    btn.classList.toggle('active', active);

    btn.setAttribute('aria-selected', String(active));

    btn.setAttribute('tabindex', active ? '0' : '-1');

    panel.classList.toggle('active', active);

  });

  document.getElementById('main-content').scrollTop = 0;

}


function toggleAcc(id) {

  const acc  = document.getElementById(id);

  const btn  = acc.querySelector('.accordion-header');

  const body = acc.querySelector('.accordion-body');

  const open = acc.classList.toggle('open');

  btn.setAttribute('aria-expanded', String(open));

  if (open) body.style.paddingTop = '12px';

}


// Keyboard arrow navigation for tab bar

document.querySelectorAll('.tab-btn').forEach((btn, i, all) => {

  btn.addEventListener('keydown', e => {

    let idx = i;

    if (e.key === 'ArrowRight') idx = (i+1) % all.length;

    else if (e.key === 'ArrowLeft') idx = (i-1+all.length) % all.length;

    else if (e.key === 'Home') idx = 0;

    else if (e.key === 'End') idx = all.length - 1;

    else return;

    e.preventDefault();

    all[idx].click();

    all[idx].focus();

  });

});
